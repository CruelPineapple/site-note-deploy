# 数组方法

# flat

能够指定扁平化层数的方法：

function myFlat(arr, depth){
  if(!Array.isArray(arr) || depth <= 0) return arr
  return myFlat([].concat(...arr), depth - 1)
}

我自己想的，感觉很不错！注意单层扁平化的操作是[].concat(...array)

又想了一下，可以改成循环：

function myFlat(arr, depth){
  let res = arr
  while(depth > 0){
    res = [].concat(...res)
    depth--
  }
  return res
}

# push

Array.prototype.myPush = function(...items){
	items.forEach((item)=>{
    this[this.length] = item
  })
  return this.length
}

# filter

Array.prototype.myFilter = function(fn){
  // 参数检测就省略吧
  const res = []
  this.forEach((item)=>{
    if(fn(item)){
      res.push(item)
    }
  })
  return res
}

# map

Array.prototype.myMap = function(fn){
  const res = []
  this.forEach((item)=>{
    res.push(fn(item))
  })
  return res
}

# forEach

filter和map其实都有thisArg功能，懒得搞了，和这里的一样

Array.prototype.myForEach = function(fn, thisArg){
  let thisRef
  if(thisArg){
    thisRef = thisArg
  }else{
    thisRef = this
  }
  for(let i = 0; i < this.length; i++){
    fn.call(thisRef, this[i], i, this)
    // 后续的i和this是因为forEach有可选参数index和arr本身，所以在这里一起提供了
  }
}
